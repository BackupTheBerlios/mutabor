193 inline static void call_actions (mutabor_box_type * box,
194  struct do_aktion * aktion_list,
195  struct interpreter_parameter_list * parameters) {
196  interpreter_parameter_list * current_parameters;
197  interpreter_parameter_list * old_parameters = box->current_parameters;
198 
199  // check whether we must allocate memory
200  if ((box->current_parameters == NULL && box->parameters == NULL)
201  || box->current_parameters->next == NULL) {
202  current_parameters = (struct interpreter_parameter_list *)
203  ymalloc(box,sizeof(struct interpreter_parameter_list));
204  current_parameters->next = NULL;
205  current_parameters->data = (int *)ycalloc(box,
206  box->file->parameter_count,
207  sizeof(int));
208  if (box->parameters == NULL) {
209  mutASSERT(box->current_parameters == NULL);
210  box->current_parameters =
211  box->parameters = current_parameters;
212  } else {
213  box->current_parameters->next = current_parameters;
214  }
215 
216  }
217  // we have a free parameter list
218  else if ( box->current_parameters == NULL) {
219  current_parameters = box->current_parameters = box->parameters;
220  } else {
221  current_parameters = box->current_parameters -> next;
222  box->current_parameters = current_parameters;
223  }
224  for (struct do_aktion * aktion = aktion_list ;
225  aktion; aktion = aktion -> next) {
226  interpreter_argument_list * arguments =
227  aktion -> secondary_arguments ?
228  aktion -> secondary_arguments :
229  aktion -> arguments;
230  size_t size;
231  if (arguments == NULL) {
232  execute_aktion (box, aktion, NULL);
233  continue;
234  }
235  size = arguments->size;
236  current_parameters -> size = size;
237  mutASSERT(size <= box->file->parameter_count);
238  for (size_t i = 0 ; i < size; i++) {
239  switch (arguments->types[i]) {
240  case mutabor_argument_integer:
241  current_parameters->data[i] = arguments->data[i].constant;
242  break;
243  case mutabor_argument_parameter:
244  current_parameters->data[i]
245  = parameters->data[arguments->data[i].index];
246  break;
247  case mutabor_argument_distance:
248  current_parameters->data[i]
249  = box->distance;
250  break;
251  case mutabor_argument_anchor:
252  current_parameters->data[i]
253  = box->tonesystem->anker;
254  break;

256  current_parameters->data[i] = -1;
257  mutASSERT(arguments->types[i] != mutabor_argument_invalid);

261  }

653  void change_anker(mutabor_box_type * box, int neu)
654  {
655  tone_system * tonsys = box->tonesystem;
656  tone_system * free_tonesystem =
657  box->tonesystem == box->tonesystem_memory ?
658  &(box->tonesystem_memory[1]) :
659  &(box->tonesystem_memory[0]);
660  int i;
661  TRACE;
662  /* update the distance parameter for logic calls */
663  box->distance -= neu-tonsys->anker;
664 
665  while ( box->distance < 0 )
666  box->distance += tonsys->breite;
667 
668  box->distance %= tonsys->breite;
669 
670  /* update the tone system */
671  while (neu< MUTABOR_KEYRANGE_MIN_KEY)
672  neu += tonsys->breite;
673 
674  while (neu> MUTABOR_KEYRANGE_MAX_KEY)
675  neu -= tonsys->breite;
676 
677  free_tonesystem->ton[0]=mutabor_get_note_frequency(neu,tonsys);
678 
679  for (i=1; i<tonsys->breite; i++)
680  free_tonesystem->ton[i] =
681  free_tonesystem->ton[0] + tonsys->ton[i]-tonsys->ton[0];
682 
683  free_tonesystem->anker=neu;
684  free_tonesystem->breite=tonsys->breite;
685  free_tonesystem->periode=tonsys->periode;
686  box->tonesystem = free_tonesystem;
