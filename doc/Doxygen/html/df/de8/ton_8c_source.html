291 static void berechne_ton_endgueltig (int k)
292 { int b;
293  double help, help2;
294 
295 #ifdef DEBUG
296  if (mutabor_debug_level)
297  fprintf (stderr,"Calculate final tone values %d ",k);
298 #endif
299  switch (toene[k]->ton_typ) {
300 
301  case ton_absolut: /* hier nichts zu tun */
302 #ifdef DEBUG
303  if (mutabor_debug_level)
304  fprintf (stderr,"Absolute tone %s: %f\n",toene[k]->name,toene[k]->u.ton_absolut.ton_wert);
305 #endif
306  break;
307 
308  case ton_komplex:
309 #ifdef DEBUG
310  if (mutabor_debug_level)
311  fprintf (stderr,"Complex tone %s:\n",toene[k]->name);
312 #endif
313  b = ton_nummer (toene[k]->u.ton_komplex.bezugston);
314  berechne_ton_endgueltig (b);
315 
316  help2 = get_wert_komplex_intervall (toene[k]->u.ton_komplex.komplex_liste);
317  help = toene[b]->u.ton_absolut.ton_wert;
318 
319  toene[k]->ton_typ = ton_absolut;

321 #ifdef DEBUG
322  if (mutabor_debug_level)
323  fprintf (stderr,"Complex tone %s: %f\n",toene[k]->name, toene[k]->u.ton_absolut.ton_wert);
324 #endif
325  break;
326 
327  default:
328  fatal_error(0,"schleife");
329  }
