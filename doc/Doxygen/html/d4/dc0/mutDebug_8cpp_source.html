147 void MutInitConsole()
148 {
149 #if __WXMSW__
150  // taken from http://www.halcyon.com/~ast/dload/guicon.htm
151  static const WORD MAX_CONSOLE_LINES = 500;
152 
153  int hConHandle;
154  long lStdHandle;
155  CONSOLE_SCREEN_BUFFER_INFO coninfo;
156  FILE *fp;
157 
158  // allocate a console for this app
159  AllocConsole();
160 
161  // set the screen buffer to be big enough to let us scroll text
162  GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE),
163  &coninfo);
164  coninfo.dwSize.Y = MAX_CONSOLE_LINES;
165  SetConsoleScreenBufferSize(GetStdHandle(STD_OUTPUT_HANDLE),
166  coninfo.dwSize);
167 
168  // redirect unbuffered STDOUT to the console
169  lStdHandle = (long)GetStdHandle(STD_OUTPUT_HANDLE);
170  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
171  fp = _fdopen( hConHandle, "w" );
172  *stdout = *fp;
173  setvbuf( stdout, NULL, _IONBF, 0 );
174 
175  // redirect unbuffered STDIN to the console
176  lStdHandle = (long)GetStdHandle(STD_INPUT_HANDLE);
177  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
178 
179  fp = _fdopen( hConHandle, "r" );
180  *stdin = *fp;
181  setvbuf( stdin, NULL, _IONBF, 0 );
182 
183  // redirect unbuffered STDERR to the console
184  lStdHandle = (long)GetStdHandle(STD_ERROR_HANDLE);
185  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
186  fp = _fdopen( hConHandle, "w" );
187  *stderr = *fp;
188  setvbuf( stderr, NULL, _IONBF, 0 );
189 
190  // make cout, wcout, cin, wcin, wcerr, cerr, wclog and clog
191  // point to console as well
192  std::ios::sync_with_stdio();

244 void mutAssertFailure(const wxChar *file,
245  int line,
246  const wxChar *func,
247  const wxChar *cond,
248  const wxChar *msg)
249 {
250 #ifdef MUTABOR_CPPUNIT
251  {
252  CPPUNIT_NS::Message mesg( "assertion failed",
253  std::string(wxString::Format(_T("%s:%d:%s\nFunction: %s; Expression: %s"),
254  file,
255  line,
256  (msg?msg:_T("")),
257  func,
258  cond).ToUTF8()));
259 
260  wxString filename(file);
261  CPPUNIT_NS::Asserter::fail(mesg,
262  CPPUNIT_NS::SourceLine(std::string(filename.ToUTF8()), line )
263  );
264  }
265 
266 #else
267  if (!std::clog.good()) MutInitConsole();
268  wxString s = wxString::Format(_("%s:%d: An assert failed in %s().\nCondition: %s\nMessage:%s\nDo you want to call the default assert handler?"),
269  file,line,func,cond,msg);
270  std::clog << (const char *) s.ToUTF8() << std::endl;
271 
272 
273  // use wxWidgets ShowAssertDialog function (it's static so we repeat its code, here.
274 
275  {
276  // this variable can be set to true to suppress "assert failure" messages
277  static bool s_bNoAsserts = false;
278 
279  wxString message;
280  message.reserve(2048);
281 
282  // here, we change format to the gcc form
283  message.Printf(wxT("%s:%d: assert \"%s\" failed"), file, line, cond);
284 
285  // add the function name, if any
286  if ( func && *func )
287  message << _T(" in ") << func << _T("()");
288 
289  // and the message itself
290  if ( msg )
291  {
292  message << _T(": ") << msg;
293  }
294  else // no message given
295  {
296  message << _T('.');
297  }
298 
299 #if wxUSE_THREADS
300  // if we are not in the main thread, output the assert directly and trap
301  // since dialogs cannot be displayed
302  if ( !wxThread::IsMain() )
303  {
304  message += wxT(" [in child thread]");
305 
306 #if defined(__WXMSW__) && !defined(__WXMICROWIN__)
307  message << wxT("\r\n");
308  OutputDebugString(message );
309 #else
310  // send to stderr
311  wxFprintf(stderr, wxT("%s\n"), message.c_str());
312  fflush(stderr);
313 #endif
314  // He-e-e-e-elp!! we're asserting in a child thread
315  wxTrap();
316  }
317  else
318 #endif // wxUSE_THREADS
319 
320  if ( !s_bNoAsserts )
321  {
322  // send it to the normal log destination
323  wxLogDebug(_T("%s"), message.c_str());
324 
325 #ifdef __WXDEBUG__
326  if ( wxGetApp().GetTraits() )
327  {
328  // delegate showing assert dialog (if possible) to that class
329  s_bNoAsserts = wxGetApp().GetTraits()->ShowAssertDialog(message);
330  }
331  else // no traits object
332 #endif
333  {
334  // fall back to the function of last resort
335  s_bNoAsserts = DoShowAssertDialog(message);
336  }
337  }
338  }
339 #endif
340 }
