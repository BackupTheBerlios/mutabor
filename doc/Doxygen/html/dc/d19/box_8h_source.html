215 inline mutabor_note_type * mutabor_find_key_in_box(mutabor_box_type * box, size_t index) {
216  mutabor_key_index_type *plane_ptr;
217  size_t plane;
218  mutabor_check_key_count(box);
219  if (!box->key_count || index == MUTABOR_NO_NEXT) return NULL;
220  plane = index >> MUT_BOX_MAX_KEY_INDEX_LOG;
221  plane_ptr = &(box->current_keys);
222  while (plane-- && plane_ptr) plane_ptr = plane_ptr->next;
223  if (!plane_ptr) {
224  return NULL;
225  }
226  return &(plane_ptr->key[index & MUT_BOX_MAX_KEY_INDEX]);
227 }

257 inline void mutabor_delete_key_in_box(mutabor_box_type * box, size_t index) {
258  mutabor_note_type *last_key, *current_key, *next_key;
259 
260  if (!box->key_count) return;
261 
262  current_key = mutabor_find_key_in_box(box,index);
263  if (current_key == NULL) {
264  UNREACHABLE;
265  return;
266  }
267 
268  if (!index) {
269 
270  if (current_key -> next != MUTABOR_NO_NEXT) {
271  next_key = mutabor_find_key_in_box(box,current_key->next);
272  if (current_key->next == box->last_key)
273  box -> last_key = 0;
274  *(current_key) = *(next_key);
275  next_key->next = MUTABOR_NO_NEXT;
276  next_key->userdata = NULL;
277  box->key_count--;
278 
279  } else {
280  mutASSERT(box->key_count == 1);
281  current_key->userdata = NULL;
282  box->key_count = 0;
283  }
284  return;
285  }

314 inline mutabor_note_type * mutabor_create_key_in_box (mutabor_box_type * box) {
315  mutabor_note_type * last, *new_key = NULL;
316  mutabor_key_index_type *plane;
317  mutabor_key_index_type *oldplane;
318  size_t index = 0;
319  size_t planeidx = 0;
320  size_t curridx = 0;
321  bool found = false;
322 
323  mutASSERT(box);
324 
325  if (!box->key_count) {
326  box->key_count = 1;
327  mutASSERT(box->current_keys.key[0].userdata == NULL);
328  mutASSERT(box->current_keys.key[0].next == MUTABOR_NO_NEXT);
329  return &(box->current_keys.key[0]);

333

335  if (last == NULL) {

349  index = i + curridx;

351  found = true;
352  break;
353  }
354  }
355  if (found) break;
356  planeidx++;
357  oldplane = plane;
358  plane = plane->next;
359  }
360  if (new_key == NULL) {

362  if (plane == NULL) {
363  mutabor_check_key_count(box);
364  return NULL;
365  }
366  mutabor_initialize_keyplane(plane);
367  new_key = &(plane->key[0]);
368  index = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
369  }
370 
371  box -> last_key = last -> next = index;
372  new_key -> next = MUTABOR_NO_NEXT;
373  box->key_count++;
374  mutabor_check_key_count(box);
375  return new_key;
