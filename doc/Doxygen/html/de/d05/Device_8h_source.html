136  int set_controller(size_t number, int8_t data) {
137 
138  if (controller.size() <= number)
139  controller.resize(number +1);
140  int retval = controller[number];
141  int param = -1;
142  DEBUGLOG(midiio,_T("ctrl: %d, %d => %d"),(int)number,(int)retval,(int)data);
143  controller[number] = data;
144  switch (number) {
145  case midi::NON_REGISTERED_PARAMETER_FINE:
146  case midi::NON_REGISTERED_PARAMETER_COARSE:
147  data_is_rpn = false;
148  break;
149  case midi::REGISTERED_PARAMETER_FINE:
150  case midi::REGISTERED_PARAMETER_COARSE:
151  data_is_rpn = true;
152  break;
153  case midi::DATA_ENTRY_COARSE:
154  case midi::DATA_ENTRY_FINE:
155  case midi::DATA_BUTTON_INCREMENT:
156  case midi::DATA_BUTTON_DECREMENT:
157  if (!data_is_rpn
158  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != -1
159  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != -1
160  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != 0x7f
161  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != 0x7f) {
162  param = 0x20000
163  | controller[midi::NON_REGISTERED_PARAMETER_COARSE] << 8
164  | controller[midi::NON_REGISTERED_PARAMETER_FINE];
165  } else if (data_is_rpn
166  && controller[midi::REGISTERED_PARAMETER_FINE] != -1
167  && controller[midi::REGISTERED_PARAMETER_COARSE] != -1
168  && controller[midi::REGISTERED_PARAMETER_FINE] != 0x7f
169  && controller[midi::REGISTERED_PARAMETER_COARSE] != 0x7f) {
170  param = 0x10000
171  | controller[midi::REGISTERED_PARAMETER_COARSE] << 8
172  | controller[midi::REGISTERED_PARAMETER_FINE];
173  }
174  if (param != -1) {
175  if (controller.size() <= (size_t) param)
176  controller.resize(param+1);
177  switch (number) {
178  case midi::DATA_BUTTON_INCREMENT:
179  controller[param]++;
180  break;
181  case midi::DATA_BUTTON_DECREMENT:
182  controller[param]--;
183  break;
184  case midi::DATA_ENTRY_COARSE:
185  case midi::DATA_ENTRY_FINE:
186  int value = (controller[midi::DATA_ENTRY_COARSE]) << 8
187  || (controller[midi::DATA_ENTRY_COARSE]);
188  if (!(value & 0x8080))
189  controller[param] = value;
190  }
191  }
192  break;
193  }
194 
195 
196 
197  bool found = false;
198  size_t index;
199  for (size_t i = 0; i < (looped?controller.size():first_unchanged);i++) {
200  if (controller_changed[i] == (int)number) {
201  controller_vector :: iterator beg = controller_changed.begin();
202  std::rotate(beg + i,beg+i+1 ,beg +first_unchanged);
203  found = true;
204  index = i;
205  break;
206  }
207  }
208  if (!found)
209  index = first_unchanged++;
210 
211  controller_changed[index] = (param == -1 ? number : param);
212 
213  return retval;
214  }

217 
218  switch (message) {
219  case midi::PITCH_BEND_SENSITIVITY:
220  return -1;
221  case midi::NON_REGISTERED_PARAMETER_FINE:
222  case midi::NON_REGISTERED_PARAMETER_COARSE:
223  if (controller[midi::NON_REGISTERED_PARAMETER_FINE] != -1
224  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != -1
225  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != 0x7f
226  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != 0x7f) {
227  return 0x20000
228  | controller[midi::NON_REGISTERED_PARAMETER_COARSE] << 8
229  | controller[midi::NON_REGISTERED_PARAMETER_FINE];
230  }
231  return -1;
232  case midi::REGISTERED_PARAMETER_FINE:
233  case midi::REGISTERED_PARAMETER_COARSE:
234  if (controller[midi::REGISTERED_PARAMETER_FINE] != -1
235  && controller[midi::REGISTERED_PARAMETER_COARSE] != -1
236  && controller[midi::REGISTERED_PARAMETER_FINE] != 0x7f
237  && controller[midi::REGISTERED_PARAMETER_COARSE] != 0x7f)
238  return get_index(0x10000
239  | controller[midi::REGISTERED_PARAMETER_COARSE] << 8
240  | controller[midi::REGISTERED_PARAMETER_FINE]);
241  return -1;
242  case midi::DATA_ENTRY_COARSE:
243  case midi::DATA_ENTRY_FINE:
244  case midi::DATA_BUTTON_INCREMENT:
245  case midi::DATA_BUTTON_DECREMENT:
246  if (!data_is_rpn

275  controller_vector::const_iterator retval = controller_changed.begin();

280  controller_vector::const_iterator get_next_changed_controller(const ChannelData & other,
281  controller_vector::const_iterator last) const {
282  mutASSERT(controller_changed.begin() <= last);
283  mutASSERT(last < controller_changed.end());
284  controller_vector::const_iterator retval = last;
285  do {
286  retval++;
287  } while (retval != controller_changed.end() && (*retval != -1)
288  && (size_t)(*retval) < other.controller.size()
289  && controller[*retval] == other.controller[*retval]);
290  return retval;
291  }
